CREATE TABLE FinancialAppLogs (
    id INT IDENTITY(1,1) PRIMARY KEY,
    application_name NVARCHAR(100),
    date DATE,
    time_stamp TIME,
    log_level NVARCHAR(20),
    servernames NVARCHAR(100),
    severity NVARCHAR(20),
    detailed_stack_trace NVARCHAR(MAX),
    error_message NVARCHAR(MAX)
);
